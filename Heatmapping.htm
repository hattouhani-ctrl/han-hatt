Easy Integration

Add heatmap tracking to your website in minutes

integration.js 
                            
<!-- Add this to your website's <head> -->
<script src="https://cdn.heatmaptracker.com/v1/heatmap.min.js"></script>
<script>
  // Initialize with your API key
  HeatmapTracker.init({
    apiKey: 'your-api-key',
    clickTracking: true,
    hoverTracking: true,
    scrollTracking: true
  });
</script>
